class Solution {
public:
    int numSubarrayProductLessThanK(vector<int>& nums, int k) {
      int n= nums.size();
      int prod =1, count =0;
      if(k<=1) return 0;
      for(int i=0, j=i;j<n;j++){
         prod *= nums[j];
        while(prod>=k){
            prod /= nums[i];
            i++;
        }
        count += (j-i+1);
      }
      return count;
    }
};
