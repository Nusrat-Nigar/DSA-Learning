class Solution {
public:
    bool isPalindrome(int x) {
        long long rem=0,sum=0;
       
        if(x>=0)
        {
            long long original = x;
            while(x>0)
            {
               rem=x%10;
               sum=(sum*10)+rem;
               x=x/10;
            }
            if(sum==original)
             return true;
            else
             return false;
        }
        else
        return false;
    }
};
