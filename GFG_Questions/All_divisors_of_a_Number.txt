class Solution {
  public:
    void print_divisors(int n) {
        // Code here.
      vector<int> res;
      for(int i=1;i*i<=n;i++){
          if(n%i == 0){
              res.push_back(i);
              if(i != (n/i)){
                  res.push_back(n/i);
              }
          }
      }
      
      sort(res.begin(), res.end());
      for(int i=0;i<res.size();i++)
      {
          cout<<res[i]<<" ";
      }
     }
      
};
