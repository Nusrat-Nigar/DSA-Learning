class Solution {
  public:
  
  bool reverseStr(string& s, int l, int r){
      if(l>=r)
      return true;
      
      if(s[l]!=s[r])
      return false;
      
      return reverseStr(s,l+1,r-1);
      
      
  }
    bool isPalindrome(string& s) {
        // code here
        
        
        return reverseStr(s,0,s.size()-1);
    }
 };
